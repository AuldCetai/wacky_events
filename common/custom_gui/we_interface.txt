##we_wackiness_selector

custom_button = {
	name = we_wackiness_selector
    potential = {} 
  	trigger = { always = yes}
  	effect = { 
		IF = {
			limit = {
				OR = {
					has_global_flag = we_vanilla_plus
					OR = {
								NOT = {
									has_global_flag = we_vanilla_plus
									has_global_flag = we_occasional_mystery
									has_global_flag = we_here_be_dragons
											}
								}
					}
				}
				
				set_global_flag = we_occasional_mystery
				clr_global_flag = we_vanilla_plus
				}
		ELSE_IF = {  
			limit = {
				has_global_flag = we_occasional_mystery
				}
				set_global_flag = we_here_be_dragons
				clr_global_flag = we_occasional_mystery
				}
		ELSE_IF = { 
			limit = {
				has_global_flag = we_here_be_dragons
				}
				set_global_flag = we_vanilla_plus
				clr_global_flag = we_here_be_dragons
				}
		ELSE = { set_global_flag = we_vanilla_plus }
		}	
	frame = {
			number = 1
			trigger = {has_global_flag = we_vanilla_plus}
			}
	frame = {
			number = 2
			trigger = {has_global_flag = we_occasional_mystery}
			}
	frame = {
			number = 3
			trigger = {has_global_flag = we_here_be_dragons}
			}
				
				
}

											}
								}
					}

			limit = {
				has_global_flag = we_wackiness_medium
				}
				clr_global_flag = we_wackiness_medium

				}
		ELSE_IF = { 
			limit = {
				has_global_flag = we_wackiness_everyday
				}
				clr_global_flag = we_wackiness_everyday
				set_global_flag = we_wackiness_rare
				}
		ELSE = {set_global_flag = we_wackiness_rare}
		}	
	frame = {
			number = 1
			trigger = {has_global_flag = we_wackiness_rare}
			}
	frame = {
			number = 2
			trigger = {has_global_flag = we_wackiness_medium}
			}
	frame = {
			number = 3
			trigger = {has_global_flag = we_wackiness_everyday}
			}
				
				
}
